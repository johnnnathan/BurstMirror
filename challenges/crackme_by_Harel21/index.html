def username_handler(username):
    length = len(username)
    length_int = length

    if (length_int - 1) % 2 == 0:
        length_int = length_int - 1
        if length_int > 2:
            length_int = (ord(username[2]) * 0x240e) // 2 + 1
    else: 
        length_int = (length_int - 1) * 0x9d9fd
        if length_int < 0:
            length_int += 3 
        length_int = ((length_int >> 2) * 0xd) // 7 

    return length_int
